\newcommand{\sectiontitle}[1]{%
  \vspace{1.05ex}%
  {\large\bfseries\MakeUppercase{#1}}\par
  \vspace{-0.45ex}{\rule{\linewidth}{0.5pt}}\par\vspace{0.45ex}
}

  {\small #1}\par\vspace{0.8ex}
}
\newcommand{\projectheading}[2]{%
  {\bfseries\small #1}\par
  {\small\color{MutedText}#2}\par
}
\newcommand{\spacevv}{\vspace{1.1ex}}
\newcommand{\skillline}[2]{%
  {\bfseries #1}\par
  #2\par\vspace{0.75ex}
}

\begin{multicols}{2}

\sectiontitle{\VAR{labels.summary | default('Profile') | latex_escape}}
\sectioncontent{\VAR{summary | latex_escape}}

\BLOCK{if experience}
\sectiontitle{\VAR{labels.experience | default('Experience') | latex_escape}}
\BLOCK{for exp in experience}
\keptentrybegin
{\bfseries\small \VAR{exp.company | latex_escape}} -- {\small \VAR{exp.role | latex_escape}}\par
{\small\color{MutedText}\VAR{exp.date | latex_escape}}\par
\BLOCK{if exp.description}
{\small\itshape \VAR{exp.description | latex_escape}}\par
\BLOCK{endif}
\BLOCK{if exp.bullets}
\begin{itemize}
\BLOCK{for bullet in exp.bullets}
\item \VAR{bullet | latex_escape}
\BLOCK{endfor}
\end{itemize}
\BLOCK{endif}
\keptentryend
\BLOCK{if not loop.last}
\spacevv
\BLOCK{endif}
\BLOCK{endfor}
\BLOCK{endif}

\BLOCK{if education}
\sectiontitle{\VAR{labels.education | default('Education') | latex_escape}}
\BLOCK{for edu in education}
\keptentrybegin
\textbf{\VAR{edu.degree | latex_escape}}\BLOCK{if edu.major} in \VAR{edu.major | latex_escape}\BLOCK{endif}\par
\VAR{edu.school | latex_escape}\par
\textcolor{MutedText}{\VAR{edu.date | latex_escape}}\par
\BLOCK{if edu.thesis}
\textit{Thesis: \VAR{edu.thesis | latex_escape}}\par
\BLOCK{endif}
\keptentryend
\BLOCK{if not loop.last}
\spacevv
\BLOCK{endif}
\BLOCK{endfor}
\BLOCK{endif}

\BLOCK{if contact.github or contact.linkedin}
\sectiontitle{Links}
\sectioncontent{
\BLOCK{if contact.github}
\href{\VAR{contact.github}}{\VAR{contact.github | display_url | latex_escape}}\\
\BLOCK{endif}
\BLOCK{if contact.linkedin}
\href{\VAR{contact.linkedin}}{\VAR{contact.linkedin | display_url | latex_escape}}\\
\BLOCK{endif}
}
\BLOCK{endif}

\BLOCK{if skills}
\sectiontitle{\VAR{labels.skills | default('Skills') | latex_escape}}
\sectioncontent{
\BLOCK{for category, items in skills.items()}
\skillline{\VAR{category | latex_escape}}{\VAR{items | map('latex_escape') | join(', ')}}
\BLOCK{endfor}
}
\BLOCK{endif}

\BLOCK{if languages}
\sectiontitle{\VAR{labels.languages | default('Languages') | latex_escape}}
\sectioncontent{
\BLOCK{for lang in languages}

\BLOCK{endfor}
}
\BLOCK{endif}

\BLOCK{if certifications}
\sectiontitle{\VAR{labels.certifications | default('Honors \\& Awards') | latex_escape}}

\BLOCK{for cert in certifications}

\BLOCK{if cert.description}
\VAR{cert.description | latex_escape}\\
\BLOCK{endif}
\BLOCK{endfor}
}
\BLOCK{endif}

\end{itemize}
\BLOCK{endif}
\keptentryend
\BLOCK{if not loop.last}
\spacevv
\BLOCK{endif}
\BLOCK{endfor}
\BLOCK{endif}
